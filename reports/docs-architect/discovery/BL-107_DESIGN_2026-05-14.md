### 1.5 G19 channel-add variant (pure, no Telegram I/O)

В `src/core/services/gates/owner_gates.py`:

```python
async def check_g19_channel_add(
    session: AsyncSession,
    user: User,
    channel_data: ChannelAddContext,
) -> GateResult:
    settings = get_settings()
    if channel_data.member_count < settings.rkn_threshold_subscribers:
        return GateResult(
            gate=PlacementGate.G19_BLOGGER_REGISTRY_VERIFIED,
            passed=True, blocker=True, reason_code=GateReason.OK.value,
        )
    if channel_data.is_test:
        return GateResult(
            gate=PlacementGate.G19_BLOGGER_REGISTRY_VERIFIED,
            passed=True, blocker=True, reason_code=GateReason.OK.value,
        )
    # Caller populates is_trustchannelbot_admin — gate stays pure (no I/O)
    if channel_data.is_trustchannelbot_admin is None:
        return GateResult(
            gate=PlacementGate.G19_BLOGGER_REGISTRY_VERIFIED,
            passed=False, blocker=True,
            reason_code=GateReason.SUBSCRIBER_COUNT_UNKNOWN.value,
        )
    if channel_data.is_trustchannelbot_admin:
        return GateResult(
            gate=PlacementGate.G19_BLOGGER_REGISTRY_VERIFIED,
            passed=True, blocker=True, reason_code=GateReason.OK.value,
            remediation_data={"verification_method": "trustchannelbot_admin"},
        )
    return GateResult(
        gate=PlacementGate.G19_BLOGGER_REGISTRY_VERIFIED,
        passed=False, blocker=True,
        reason_code=GateReason.BLOGGER_REGISTRY_NOT_VERIFIED.value,
        remediation_url="/owner/channel-registry-evidence",
        remediation_data={"verification_method": "manual_evidence"},
    )
```

**Key design decision:** Telegram API call moved OUT of gate body. Caller invokes `verify_trustchannelbot_admin` ДО building ChannelAddContext, populates `is_trustchannelbot_admin` boolean. Gate body remains pure (fully unit-testable без bot mock).

async def check_g19(session: AsyncSession, placement: PlacementRequest) -> GateResult:
    if placement.channel is None:
        await session.refresh(placement, ["channel"])

## 2. Helper architecture — `check_gates_for_channel_add`

### 2.1 Signature + body

В `src/core/services/legal_compliance_service.py`:

```python
async def check_gates_for_channel_add(
    self,
    user: User,
    channel_data: ChannelAddContext,
) -> list[GateResult]:
    """Evaluate all gates required for owner channel-add (BL-107).

    Orchestrates user-role gates (G04/G05/G06) and channel-context
    gates (G19). Gate ordering: user-role first; if any blocker fails,
    channel-context gates skipped.

    S-48 Pattern 1: no commit/flush/rollback. Caller owns transaction.
    """
    results: list[GateResult] = []
    user_role_results = await self.check_gates_for_user_role(user, role="owner")
    results.extend(user_role_results)
    if any(r.blocker and not r.passed for r in user_role_results):
        return results  # Owner-level failure preempts channel-context check

    for gate in _CHANNEL_ADD_GATES:
        checker = _CHANNEL_CONTEXT_GATE_CHECKERS.get(gate)
        if checker is None:
            raise NotImplementedError(
                f"No channel-context gate-checker registered for {gate.name}"
            )
        results.append(await checker(self._session, user, channel_data))
    return results
```

## 3. Cross-SDK Protocol — `verify_trustchannelbot_admin`

### 3.1 Protocol definition

В `src/utils/telegram/verify_blogger_registry.py` (new file):

```python
@runtime_checkable
class TelegramAdminLister(Protocol):
    """Minimal Bot interface для admin enumeration."""
    async def get_chat_administrators(self, chat_id: int | str) -> Any:
        ...


class TrustchannelbotResolutionError(Exception):
    """Trustchannelbot ID resolution или admin enumeration failed."""
```

Protocol supports both python-telegram-bot Bot и aiogram Bot. Return types differ (`tuple` vs `list`) but both iterable, `.user.id` accessible via duck-typing.

### 3.2 Verification helper

```python
async def verify_trustchannelbot_admin(
    bot: TelegramAdminLister,
    chat_id: int | str,
) -> bool:
    """Check if Trustchannelbot is among admins."""
    try:
        trustchannelbot_id = await resolve_trustchannelbot_id(bot)
    except Exception as e:
        raise TrustchannelbotResolutionError(...) from e

    try:
        admins = await bot.get_chat_administrators(chat_id)
    except Exception as e:
        raise TrustchannelbotResolutionError(...) from e

    admin_user_ids = {_extract_user_id(m) for m in admins}
    admin_user_ids.discard(None)
    return trustchannelbot_id in admin_user_ids


def _extract_user_id(chat_member: Any) -> int | None:
    """Duck-typed .user.id extraction — handles SDK shape drift."""
    try:
        user = getattr(chat_member, "user", None)
        return int(getattr(user, "id", None) or 0) or None
    except (AttributeError, TypeError, ValueError):
        return None
```

### 3.3 Why Protocol vs ABC / Union

Structural typing — no inheritance required. Both SDKs' Bots satisfy without changes. Pyright/mypy 1.0+ infer compatibility correctly.

## 4. Lazy cache — Trustchannelbot ID resolution

### 4.1 Storage

Module-level dict + `asyncio.Lock` (NOT functools.lru_cache — needs async lock).

```python
_trustchannelbot_id_cache: dict[str, int] = {}
_trustchannelbot_cache_lock = asyncio.Lock()
```

### 4.2 Resolver

```python
async def resolve_trustchannelbot_id(bot: Any) -> int:
    settings = get_settings()
    if settings.rkn_trustchannelbot_id is not None:
        return settings.rkn_trustchannelbot_id  # env override

    username = settings.rkn_trustchannelbot_username
    if not username:
        raise TrustchannelbotResolutionError("No identifier configured")

    if username in _trustchannelbot_id_cache:
        return _trustchannelbot_id_cache[username]  # cache hit (lock-free)

    async with _trustchannelbot_cache_lock:
        if username in _trustchannelbot_id_cache:
            return _trustchannelbot_id_cache[username]  # re-check
        try:
            chat = await bot.get_chat(username)
            chat_id = int(getattr(chat, "id", None) or 0)
            if not chat_id:
                raise TrustchannelbotResolutionError("Empty chat.id")
            _trustchannelbot_id_cache[username] = chat_id
            return chat_id
        except TrustchannelbotResolutionError:
            raise
        except Exception as e:
            raise TrustchannelbotResolutionError(...) from e


def _invalidate_trustchannelbot_cache() -> None:
    """Tests-only — pytest fixture utility."""
    _trustchannelbot_id_cache.clear()
```

### 4.3 Failure modes

| Failure | Behavior |
|---|---|
| `settings.rkn_trustchannelbot_id` set | Used directly (no Telegram API call) |
| Username + ID both empty | Raise immediately (config error) |
| First call, Telegram API down | Raise TrustchannelbotResolutionError → caller catches → ChannelAddContext.is_trustchannelbot_admin=None → G19 returns SUBSCRIBER_COUNT_UNKNOWN |
| Concurrent first calls | One resolves under lock; others wait → all read cache |
| Process restart | Cache empty, next call re-resolves (one-time cost) |
